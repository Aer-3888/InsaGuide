/* Q11 : Corps de ajoutBatiment */
int ajoutBatiment(EnsBat *E, int id, Nature n, double h, char *nv){
	E->ens[E->nbat].identifiant = id;
	E->ens[E->nbat].nature = n;
	E->ens[E->nbat].hauteur = h;
	E->ens[E->nbat].nomVille = nv;

	(E->nbat)++;
	return 1;
}

#include <string.h>

void chercheDansVille(EnsBat *E, char *nv){
	for(int i=0 ; i < (E->nbat) ; i++){
		if (!strcmp(nv,E->ens[i].nomVille))
			printf("identifiant : %d | Nature : %s | Hauteur : %f | Ville : %s \n",E->ens[i].identifiant,E->ens[i].nature,E->ens[i].hauteur,E->ens[i].nomVille);
	}
}
