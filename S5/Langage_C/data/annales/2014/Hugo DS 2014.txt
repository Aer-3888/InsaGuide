#include <string.h>
int calculeScore(char * resultats);

int calculeScore(char * resultats){
	int i;
	int score=0;
	for(i=0;i<strlen(resultats);i++){
		if(resultats[i]='G') score++;
		if(resultats[i]='P') score--;
	}
	return score;
}

int lireTournoiFichier(char * nomfichier, Tournoi * tournoi){
	int i;
	char* score;
	FILE *monfichier=(FILE*) NULL;
	if((monfichier=fopen(nomFichier,"r"))==(FILE*) NULL){
		fprintf(stderr,"Erreur d'ouverture du fichier %s\n",nomFichier);
		return -1;
	}
	
	fscanf(monfichier,"%d",&(tournoi->nb_equipes));
		
	if((tournoi->equipes=(Equipe *) calloc(tournoi->nb_equipes,sizeof(Equipe)))==NULL){
		printf("Problème allocation mémoire");
		return -2;
	}
	
	for(i=0;i<tournoi->nb_equipes;i++){
		fscanf(monfichier,"%d",&(((tournoi->equipes)[i]).numero));
		fscanf(monfichier,"%s",score);
		((tournoi->equipes)[i]).score_total=calculeScore(score);
		fscanf(nomFichier,"%[^\n\r]",((tournoi->equipes)[i]).nom_equipe);
	}
	
	if(fclose(monfichier)){
		fprintf(stderr,"Problème fermeture du fichier\n");
		return -1;
	}
	return 0;
}
