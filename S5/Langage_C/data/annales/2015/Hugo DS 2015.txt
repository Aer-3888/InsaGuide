int rechercherIndice (char* titre, Livre* s,int taille){
	int i;
	for(i=0;i<taille;i++){
		if(strcmp(s[i].titre,titre)>0){
			return i;
		}
	}
	return taille;

}

Q11)

void decalerStock(Livre* s, int taille, int indice){
	int i;
	for(i=taille;i>indice,i--){
		s[i]=s[i-1];
	}
}

int ajouterLivre (Livre *s, int taille, int tailleMax, Livre *l){
	int indice;
	if(taille==tailleMax){
		printf("La bibliothèque est pleine");
		return taille;
	}
	else{
		indice=rechercherIndice(l->titre,s,taille);
		decalerStock(s,taille,indice);
		s[indice]=l;
		taille++;
		return taille;
	}
}
