	int borne_inf, borne_sup;
	int i;
	int gagnant;

	printf("Veuillez saisir votre nom\n");
	scanf("%s",nom);

	printf("%s, veuillez saisir le nombre de numéros gagnants à tirer\n",nom);
	scanf("%d",&nb_gagnants);

	printf("%s, veuillez donner le plus petit chiffre possible\n",nom);
	scanf("%d",&borne_inf);

	printf("%s, veuillez donner le plus grand chiffre possible\n",nom);
	scanf("%d",&borne_sup);

	for(i=0; i<nb_gagnants ; i++){
		srand(time(NULL));
		gagnant = ((float)rand()*(borne_sup - borne_inf+1))/RAND_MAX + borne_inf ;
		printf("Voici le numéro gagnant numéro %d : %d \n",i,gagnant);
	}

	return 0;
}

#include <stdio.h>

/* definition de la structure Livre */

int rechercherIndice(char * titre, Livre * s, int taille){
	int i = 0;
	int find = 0;
    while( i<taille && !find){
    	if(strcmp(titre,s[i]) <= 0) find=1;
    	else i++;
    }
    return i;
}

void decalerStock(Livre * s, int taille, int indice){
	for(int i=taille-1 ; i>=indice ; i--){
		s[i+1]=s[i];
	}
}

int ajouterLivre (Livre * s, int taille, int tailleMax, Livre * l){
	if (taille==tailleMax){
		printf("Le stock est déjà à son maximum. dsl mon vieux.");
		return taille;
	} else {
		int indice = rechercherIndice(l->titre,s,taille);
		decalerStock(s,taille,indice);
		s[indice]=l;
		return taille+1;
	}
}
