int compareDate(Date d1, Date d2){     /* C'est un peu dégueulasse mais bon */
	if(equalDate(d1,d2)){
		return 1;
	}
	else{
		if(d1.annee>d2.annee){
			return 2;
		}
		if(d1.annee<d2.annee){
			return 0;
		}
		if(d1.annee==d2.annee){
			if(d1.mois>d2.mois){
				return 2;
			}
			if(d1.mois<d2.mois){
				return 0;
			}
			if(d1.mois==d2.mois){
				if(d1.jour>d2.jour){
				return 2;
			}
			if(d1.jour<d2.jour){
				return 0;
			}
			}
		}
	}
}

#include <stdio.h>

#define ERROR_ALLOC 43567

Date* copyDate(Date d){
	Date * res;
	if((res=(Date*) malloc(sizeof(Date)))==NULL){
		fprintf(stderr, "Problème d'allocation mémoire\n");
		exit(ERROR_ALLOC);
	}
	res->jour=d.jour;
	res->mois=d.mois;
	res->annee=d.annee;
	return res;
}
