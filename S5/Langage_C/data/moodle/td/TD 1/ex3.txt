#include <stdio.h>
main ()
{
    /*déclaration des variables*/
	int i,n;
	int estPremier=1;
    
    /*saisie au clavier*/
	printf("Entrez la valeur de n : ");
	scanf("%d", &n);
    
    /*Test d'un nombre premier*/
	for(i=2,i<n,i++){
		if(n%i==0){
		estPremier=0;
		}
	}
    
    /*affichage du résultat*/
	if(estPremier){ printf("Le nombre %d est premier",n);
	}else{ printf("Le nombre %d n'est pas premier",n);
	}
    
    return 0;
}
